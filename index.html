import urllib.request, math, xml.etree.ElementTree as ET BASE = "https://raw.githubusercontent.com/hammerbamsen/casa-hammerby/main/gpx" files = { "cykling": [ "Andratx - clockwise.gpx", "Cala Tuent.gpx", "Coll de Femenia.gpx", "Orient - anti-clockwise.gpx", "Orient extended - anti-clockwise.gpx", "Port des Canonge.gpx", "Sobremunt.gpx", ], "vandring": [ "Binaraix.gpx", "Biniaraix loop - short.gpx", "PdS to Fornalutx.gpx", "Torra Picada.gpx", ], "lob": [ "Basin de Cuber.gpx", "Far des Cap Gros - long.gpx", ], } def haversine(lat1, lon1, lat2, lon2): R = 6371000 φ1, φ2 = math.radians(lat1), math.radians(lat2) dφ = math.radians(lat2 - lat1) dλ = math.radians(lon2 - lon1) a = math.sin(dφ/2)**2 + math.cos(φ1)*math.cos(φ2)*math.sin(dλ/2)**2 return R * 2 * math.atan2(math.sqrt(a), math.sqrt(1-a)) def parse_gpx(url): req = urllib.request.urlopen(url, timeout=30) data = req.read() root = ET.fromstring(data) ns = {'gpx': 'http://www.topografix.com/GPX/1/1'} pts = [] for trkpt in root.findall('.//gpx:trkpt', ns): lat = float(trkpt.get('lat')) lon = float(trkpt.get('lon')) ele_el = trkpt.find('gpx:ele', ns) ele = float(ele_el.text) if ele_el is not None else 0 pts.append((lat, lon, ele)) if not pts: return None, None dist = sum(haversine(pts[i][0], pts[i][1], pts[i+1][0], pts[i+1][1]) for i in range(len(pts)-1)) gain = sum(max(0, pts[i+1][2] - pts[i][2]) for i in range(len(pts)-1)) return round(dist/1000, 1), round(gain) for folder, names in files.items(): print(f"\n=== {folder} ===") for name in names: url = f"{BASE}/{folder}/{urllib.parse.quote(name)}" try: km, hm = parse_gpx(url) print(f" {name}: {km} km · {hm} hm") except Exception as e: print(f" {name}: ERROR {e}") import urllib.parse
